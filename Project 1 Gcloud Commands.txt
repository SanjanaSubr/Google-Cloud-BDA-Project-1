CREATE TABLE employee_records AS
SELECT
    employees.emp_no,
    employees.birth_date,
    employees.first_name,
    employees.last_name,
    employees.gender,
    employees.hire_date,
    titles.title,
    titles.from_date AS titles_from_date,
    titles.to_date AS titles_to_date,
    salaries.salary,
    salaries.from_date AS salaries_from_date,
    salaries.to_date AS salaries_to_date,
    departments.dept_name,
    dept_emp.from_date AS dept_emp_from_date,
    dept_emp.to_date AS dept_emp_to_date
FROM
    employees
LEFT JOIN
    titles ON employees.emp_no = titles.emp_no
LEFT JOIN
    salaries ON employees.emp_no = salaries.emp_no
LEFT JOIN
    dept_emp ON employees.emp_no = dept_emp.emp_no
LEFT JOIN
    departments ON dept_emp.dept_no = departments.dept_no;
